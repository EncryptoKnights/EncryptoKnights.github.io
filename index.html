encrypt.js
// Initialize Encrypto Knight SDK
import { EncryptoKnight } from '@encrypto-knight/sdk';

const ek = new EncryptoKnight({
  apiKey: process.env.EK_API_KEY,
  encryption: 'aes-256-gcm',
  mode: 'zero-trust'
});

// Encrypt sensitive data
const encrypted = await ek.encrypt(
  userData,
  { e2e: true }
);

// Decentralized storage
await ek.store(encrypted, {
  replicas: 3,
  region: 'auto'
});

// ✓ Done! Your data is secured.
